Программа "Спрайт"

 


echo off
chcp 1251
color 1d
title Sprayt
setlocal ENABLEdelayEdexpanSION
:1
set X=1
mode con cols=102 lines=60
FOR /F "usebackq delims=" %%i IN (Sprayt.txt) DO (
	echo.%%i
	set /a X=X+1
	if !X! gtr 57 (
		set X=1
		>nul pathping /h 1 /p 80 /q 1 /w 1 127.0.0.1
		cls
		)
	)
goto 1

Программа "Таблица умножения на C++"

 


#include <iostream>
using namespace std;

void line1(char S)
{
	cout << "  ";
	int i = 1;
	while (i <= 71)
	{
		i++;
		cout << S;

	}
	cout << endl;
}

void line2(char S)
{
	cout << "  " << S;
		int i = 1;
		do
	{
		i++;
		cout << "      " << S;
	} while (i <= 10);
	cout << endl;
}

int line3(char S, int k)
{
	cout << "  " << S;
	for (int i = 1; i <= 10; i++)
	{
		int R = i * k;
		R < 10 ? (cout << "   ") : R < 100 ? (cout << "  ") : (cout << " ")
			; cout << R << "  " << S;
	}
	cout << endl;
	return 0;
}

void main()
{
	cout << endl;
